// Add these rules to your Firebase Storage console

service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      // Allow public read access to all files
      allow read;
      
      // Allow authenticated users to write to their own folders
      match /doctor_images/{userId}/{fileName} {
        allow write: if request.auth != null && request.auth.uid == userId;
      }
      
      match /doctor_gallery/{userId}/{fileName} {
        allow write: if request.auth != null && request.auth.uid == userId;
      }
      
      // Default rule - require authentication for writing
      allow write: if request.auth != null;
    }
  }
}
